1function [ Aobs, Bobs, A, B, Ef, Opseudinv ] = Continuous_AB_Comp( delta, vx, vy, theta, C, option )
 2
 3%     lf          = 0.902;
 4%     lr          = 0.638;
 5%     m           = 196;
 6%     I           = 93;
 7%     Cf          = 17974;
 8%     Cr          = 24181;  
 9%     mu          = 1.4;
10    
11    lf          = 0.125;
12    lr          = 0.125;
13    m           = 1.98;
14    I           = 0.03;
15    Cf          = 60;
16    Cr          = 60;  
17    mu          = 0.05;
18    
19    if option == 'Invariant B'
20        
21        B = [ 0      1 ;	%[delta accel]
22             Cf/m    0 ; 
23             lf*Cf/I 0 ;
24              0      0 ;
25              0      0 ;
26              0      0]; 
27         
28    elseif option == 'Varianttt B'
29        
30        B = [ -Cf*sin(delta)/m      1 ;     %[delta accel]
31              Cf*cos(delta)/m       0 ; 
32              lf*Cf*cos(delta)/I    0 ;
33                    0               0 ;
34                    0               0 ;
35                    0               0];
36    end
37     
38
39    A11 =  -mu;   
40    A12 = (sin(delta) * Cf) / (m*vx);
41    A13 = (sin(delta) * Cf * lf) / (m*vx) + vy;
42    A22 = -(Cr + Cf * cos(delta)) / (m*vx);
43    A23 = -(lf * Cf * cos(delta) - lr * Cr) / (m*vx) - vx;
44    A32 = -(lf * Cf * cos(delta) - lr * Cr) / (I*vx);
45    A33 = -(lf * lf * Cf * cos(delta) + lr * lr * Cr) / (I*vx);
46
47    A = [ A11               A12     A13  0  0  0 ;  %[vx]
48           0                A22     A23  0  0  0 ;  %[vy]
49           0                A32     A33  0  0  0 ;  %[wz]  
50           cos(theta)  -sin(theta)  0    0  0  0 ;
51           sin(theta)   cos(theta)  0    0  0  0 ;
52           0                0       1    0  0  0 ;];
53
54 
55    Ef         = [0;
56                  0;
57                  0;
58                  0;
59                  0;
60                  0];           % Perturbation vector: Drag force made by the wind over the vehicle.
61    
62    Opseudinv   = pinv(C*Ef);                 
63    
64    Aobs = (eye(6) - Ef*Opseudinv*C)*A;
65    Bobs = (eye(6) - Ef*Opseudinv*C)*B;
66
67    
68end
